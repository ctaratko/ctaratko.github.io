standardize <- function(df, year_label) {

  df <- clean_names(df)

  # Identify possible fill column
  fill_col <- names(df)[str_detect(names(df), "fill")][1]

  # Only rename IF needed
  if (fill_col != "number_of_tubes_first_filled") {
    df <- df %>%
      rename(number_of_tubes_first_filled = all_of(fill_col))
  }

  # Same logic for week column
  week_col <- names(df)[str_detect(names(df), "week")][1]

  if (week_col != "week") {
    df <- df %>%
      rename(week = all_of(week_col))
  }

  df %>%
    mutate(
      year = year_label,
      number_of_tubes_first_filled =
        replace_na(number_of_tubes_first_filled, 0)
    ) %>%
    select(year, week, number_of_tubes_first_filled)
}
